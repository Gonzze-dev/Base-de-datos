d) SELECT * FROM empleado WHERE sueldo > 20000;

e) SELECT (SUM(sueldo)/COUNT(*)) AS Promedio_Sueldo 
FROM empleado
INNER JOIN trabajapara ON empleado.dni = trabajapara.dni
INNER JOIN departamento on trabajapara.codigo = departamento.codigo
WHERE departamento.nombre = 'COMPUTOS';

f) SELECT SUM(horas) AS horas_produccion 
FROM empleado
INNER JOIN trabajapara ON empleado.dni = trabajapara.dni
INNER JOIN departamento on trabajapara.codigo = departamento.codigo
WHERE departamento.nombre = 'PRODUCCION';

g) SELECT empleado.nombre
FROM empleado
INNER JOIN trabajapara ON empleado.dni = trabajapara.dni
INNER JOIN departamento on trabajapara.codigo = departamento.codigo
WHERE departamento.nombre = 'DEPOSITO'
    AND horas >= 25;
